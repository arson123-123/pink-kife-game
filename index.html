<script>

const canvas = document.getElementById('game');

const ctx = canvas.getContext('2d');

const scoreEl = document.getElementById('score');

const startScreen = document.getElementById('startScreen');

 

let score = 0;

let gameOver = false;

let gameStarted = false;

 

// 刀座大小改回原圖 100%

const holder = { x: 140, y: 540, width: 140, height: 180 };

 

let knives = [];

let knifeSpeed = 4.5;

let spawnRate = 90;

let frame = 0;

let acceleration = 0;

 

// 最新圖片 links

const knifeImg = new Image();

knifeImg.crossOrigin = "anonymous";

knifeImg.src = 'https://i.imgur.com/RqnGcmv.png';

 

const holderImg = new Image();

holderImg.crossOrigin = "anonymous";

holderImg.src = 'https://i.imgur.com/Co4zydY.png';

 

const logoImg = new Image();

logoImg.crossOrigin = "anonymous";

logoImg.src = 'https://i.imgur.com/kIqy3LI.jpeg';

 

let knifeLoaded = false;

let holderLoaded = false;

let logoLoaded = false;

 

knifeImg.onload = () => { knifeLoaded = true; };

holderImg.onload = () => { holderLoaded = true; };

logoImg.onload = () => { logoLoaded = true; };

 

// 畫真刀（12.5% 大小，垂直）

function drawKnife(k) {

  if (knifeLoaded) {

    ctx.save();

    ctx.translate(k.x + 22, k.y + 14);

    ctx.drawImage(knifeImg,

      -knifeImg.width * 0.0625,

      -knifeImg.height * 0.0625,

      knifeImg.width * 0.125,

      knifeImg.height * 0.125);

    ctx.restore();

  } else {

    ctx.fillStyle = '#d3d3d3'; ctx.fillRect(k.x, k.y, 22, 5);

    ctx.fillStyle = '#ff99cc'; ctx.fillRect(k.x + 22, k.y - 2, 7, 7);

  }

}

 

// 畫真刀座（100% 原大小）

function drawHolder() {

  if (holderLoaded) {

    ctx.drawImage(holderImg, holder.x, holder.y, holder.width, holder.height);

  } else {

    ctx.fillStyle = '#f5e8d3'; ctx.fillRect(holder.x + 10, holder.y + 130, 120, 50);

    ctx.fillStyle = '#ff99cc'; ctx.fillRect(holder.x, holder.y, holder.width, 130);

    ctx.fillStyle = '#ff6699'; ctx.fillRect(holder.x + 20, holder.y + 10, 40, 100); ctx.fillRect(holder.x + 80, holder.y + 10, 40, 100);

  }

}

 

// 畫真 logo

function drawLogo() {

  if (logoLoaded) {

    ctx.drawImage(logoImg, 20, 20, 110, 110);

  } else {

    ctx.save(); ctx.translate(50, 50); ctx.scale(0.9, 0.9); ctx.fillStyle = '#c2185b';

    ctx.beginPath(); ctx.arc(10, 25, 15, 0, Math.PI*2); ctx.fill(); ctx.fillRect(0, 35, 15, 60);

    ctx.beginPath(); ctx.arc(50, 25, 15, 0, Math.PI*2); ctx.fill(); ctx.fillRect(40, 35, 15, 60);

    ctx.fillRect(10, 40, 50, 12);

    ctx.restore();

  }

}

 

function startGame() {

  startScreen.style.display = 'none';

  gameStarted = true;

  gameLoop();

}

 

function gameLoop() {

  if (!gameStarted) return;

  if (gameOver) {

    ctx.fillStyle = 'rgba(0,0,0,0.7)'; ctx.fillRect(0,0,canvas.width,canvas.height);

    ctx.fillStyle = '#fff'; ctx.font = 'bold 60px Arial'; ctx.textAlign = 'center';

    ctx.fillText('Game Over', canvas.width/2, canvas.height/2 - 80);

    ctx.font = '40px Arial'; ctx.fillText(分數: ${score}, canvas.width/2, canvas.height/2);

    ctx.font = '28px Arial'; ctx.fillText('親親無限接刀大賽～ 重新載入再戰！', canvas.width/2, canvas.height/2 + 100);

    return;

  }

 

  ctx.clearRect(0,0,canvas.width,canvas.height);

 

  const grad = ctx.createLinearGradient(0,0,0,canvas.height);

  grad.addColorStop(0, '#fff0f5'); grad.addColorStop(1, '#ffcce6');

  ctx.fillStyle = grad; ctx.fillRect(0,0,canvas.width,canvas.height);

 

  drawLogo();

 

  frame++;

 

  acceleration += 0.004;

  knifeSpeed = 4.5 + acceleration + (score * 0.12);

  spawnRate = Math.max(25, 90 - Math.floor(acceleration * 4) - Math.floor(score * 1.2));

 

  if (frame % spawnRate === 0) {

    const num = (score > 15 && Math.random() < 0.3) ? 2 : 1;

    for (let i = 0; i < num; i++) {

      knives.push({ x: Math.random() * (canvas.width - 60) + 10, y: -40 - i * 10 });

    }

  }

 

  for (let i = knives.length - 1; i >= 0; i--) {

    let k = knives[i];

    k.y += knifeSpeed;

 

    drawKnife(k);

 

    // 碰撞範圍（刀小，刀座正常大小）

    if (k.y + 28 > holder.y && k.y < holder.y + holder.height &&

        k.x + 38 > holder.x && k.x + 8 < holder.x + holder.width) {

      score++;

      scoreEl.textContent = score;

      knives.splice(i, 1);

      continue;

    }

 

    if (k.y > canvas.height + 50) {

      gameOver = true;

    }

  }

 

  drawHolder();

 

  requestAnimationFrame(gameLoop);

}

 

// 控制

function moveHolder(clientX) {

  if (!gameStarted || gameOver) return;

  const rect = canvas.getBoundingClientRect();

  holder.x = clientX - rect.left - holder.width / 2;

  holder.x = Math.max(0, Math.min(canvas.width - holder.width, holder.x));

}

 

canvas.addEventListener('mousemove', e => moveHolder(e.clientX));

canvas.addEventListener('touchmove', e => { e.preventDefault(); moveHolder(e.touches[0].clientX); }, {passive: false});

 

window.addEventListener('keydown', e => {

  if (e.key === 'ArrowLeft' || e.key === 'a') holder.x -= 45;

  if (e.key === 'ArrowRight' || e.key === 'd') holder.x += 45;

  holder.x = Math.max(0, Math.min(canvas.width - holder.width, holder.x));

});
